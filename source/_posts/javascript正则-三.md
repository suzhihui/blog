# 常用的正则表达式
- 验证是否为有效数字
```js
/**
 * 分析
 * 1.可能出现 + - 号，也可能不出现
 * 2.一位0-9都可以，多位首位不能是0
 * 3.小数部分可能有可能没有，一旦有后面必须有小数点+数字
 */
let reg = /^(+|-)?(\d|([1-9]\d+))(\.\d+)?$/
```

- 真实姓名

```js
/**
 * 尼古拉斯丶 赵四
 * 1.汉字 /^[\u4E00-\u9FA5]$/
 * 2.名字长度
 * 3.可能有译名丶汉字
 */
let reg = /^[\u4E00-\u9FA5]{2,10}(丶[\u4E00-\u9FA5]{2,10}){0,2}$/
```

- 验证邮箱

```js
let reg = /^\w+((-\w+)|(\.\w+))*@[A-Za-z0-9]+((\.|-)[pA-Za-z0-9]+)*\.[A-Za-z0-9]+$/
```

- 身体证号码

```js
/**
 * 18位
 * 最后 一位可能有x
 * 
 * 前6位：省市县
 * 中间8位：年月日
 * 最后四位
 *  最后一位 => X或者数字
 *  倒数第二位 => 偶数 女  奇数 男
 *  其余的是经过算法算出来的
 */
let reg = /^\d{17}(\d|X)$/
// 小括号分组捕获
let reg2 = /^(\d{6})(\d{4})(\d{2})(\d{2})\d{2}(\d)(\d|x)$%/
reg2.exec("130828199012040517")
// ["130828199012040517", "130828", "1990", "12", "04", "1", "7", index: 0, input: "130828199012040517", groups: undefined]
```
